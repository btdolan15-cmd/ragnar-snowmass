<script>

let runners = ["Bryce","Skylar","Alex","Maggie","Nick","Billy","Kyle","Taylor"];

let state = JSON.parse(localStorage.getItem("ragnarState")) || {

  index: -1,

  startTime: null,

  history: []

};


function save() {

  localStorage.setItem("ragnarState", JSON.stringify(state));

}


function updateUI() {

  if (state.index >= 0) {

    document.getElementById("runner").innerText = runners[state.index];

    document.getElementById("start").innerText = new Date(state.startTime).toLocaleTimeString();


    let loopTime = document.getElementById("loopSelect").value;

    document.getElementById("loop").innerText = loopTime + " min";


    let eta = new Date(state.startTime + loopTime * 60000);

    document.getElementById("eta").innerText = eta.toLocaleTimeString();


    let next = runners[(state.index + 1) % runners.length];

    document.getElementById("next").innerText = next;

  }

}


function startRun() {

  state.index = (state.index + 1) % runners.length;

  state.startTime = Date.now();


  let next = runners[(state.index + 1) % runners.length];


  alert(runners[state.index] + " is running!\n" + next + " you're on deck!");


  save();

  updateUI();

}


function finishRun() {

  let finishTime = Date.now();

  let duration = (finishTime - state.startTime) / 60000;


  state.history.push({

    runner: runners[state.index],

    time: duration.toFixed(1)

  });


  alert(runners[state.index] + " finished in " + duration.toFixed(1) + " min");


  save();

}


function shareLocation() {

  if (navigator.geolocation) {

    navigator.geolocation.getCurrentPosition(pos => {

      let coords = pos.coords.latitude + "," + pos.coords.longitude;

      alert("Share this:\nhttps://maps.google.com/?q=" + coords);

    });

  } else {

    alert("GPS not supported");

  }

}


updateUI();
